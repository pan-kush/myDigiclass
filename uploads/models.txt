CREATE TABLE IF NOT EXISTS college(
	clg_id text PRIMARY KEY,
	name text NOT NULL,
	password text
);
CREATE TABLE IF NOT EXISTS user(
	user_id text PRIMARY KEY,
	clg_id text,
	name text NOT NULL,
	password text,
	occ text,
	roll text
);
CREATE TABLE IF NOT EXISTS class(
	cls_id text PRIMARY KEY,
	desc text,
	creator text,
	logged boolean
);
CREATE TABLE IF NOT EXISTS urc(
	user_id text,
	cls_id text,
	privilage boolean
);
CREATE TABLE IF NOT EXISTS assignment(
	ass_id text PRIMARY KEY,
	cls_id text,
	ass_name text NOT NULL,
	deadline text,
	desc text,
	files text
);
CREATE TABLE IF NOT EXISTS ura(
	user_id text,
	ass_id text,
	sub_time text,
	content text
);

CREATE TABLE IF NOT EXISTS cra(
	cls_id text,
	ass_id text
);
CREATE TABLE IF NOT EXISTS chat(
	cls_id text,
	user_id text,
	chat_id text PRIMARY KEY,
	dtime text,
	text text,
	file text
);
CREATE TABLE IF NOT EXISTS clrch(
	cls_id text,
	chat_id text
);
